NFA $\mathcal M_i$ mit $L(\mathcal M_i)=L(r_i)$ an.
\begin{enumerate}  \item $r_1 = (ab)^*$  \item $r_2 = a(b \mid c)a^* \mid a^*$% \item $r_3 = (bb\mid cc^*)^*$\end{enumerate}

aus der Vorlesung an.
\end{exercise}
