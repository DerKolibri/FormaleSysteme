      (     a \vee      b) \wedge
      (\neg a \vee  c ) \wedge
      ( a \vee \neg c) \wedge {}
      ( b \vee \neg d) \wedge
      (\neg b  \vee d) \wedge
      (\neg c \vee d) \wedge
      (c \vee \neg d)
  \end{gather*}
 \end{enumerate}
